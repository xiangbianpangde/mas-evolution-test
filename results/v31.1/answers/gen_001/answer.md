def calculate_quantum_readiness(financial_institution_data):
    """
    计算金融机构量子就绪度得分
    
    参数: 包含各维度评估数据的字典
    返回: 0-100的 Readiness Score
    """
    total_score = 0
    
    for dimension, config in quantum_readiness_score["dimensions"].items():
        dimension_score = 0
        for metric, weight in config["metrics"].items():
            # 实际应用中需要从机构数据中获取评分
            metric_score = financial_institution_data[dimension][metric]
            dimension_score += metric_score * weight
        
        total_score += dimension_score * config["weight"]
    
    return total_score * 20  # 转换为0-100分制
